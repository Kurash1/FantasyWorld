new class hathkor_order = {
	id = `hathkor_order_{this:id}`
	list = hathkor_orders
	args = {
		string name
		country tag
		named_modifier modifier
	}
	on_create = {
		this:tag:history +=! {
			quick_country_modifier = {
				id = "{this:id}"
				name = "{this:name}"
				desc = "During the times of the empires founding, the most elite of noble warriors founded a knightly order called the Hatikite Order. For their service the Emperor granted them holdings in Hathkor. The '{this:name}' is a descendant of that original knightly order."
				modifier = this:modifier
			}
		}
		new mercenary_company `{this:id}` ={
			name = "{this:name}"
			desc = "During the times of the empires founding, the most elite of noble warriors founded a knightly order called the Hatikite Order. For their service the Emperor granted them holdings in Hathkor. The '{this:name}' is a descendant of that original knightly order."
			regiments_per_development = 0.04
			modifier = this:modifier
			cavalry_weight = 0.2
			artillery_weight = 0.1
			cost_modifier = 1.0
			trigger =* {
				NOT->`exists` = *this:tag:tag*
			}
		}
	}
}
